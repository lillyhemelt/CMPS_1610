import java.util.Scanner;

public class Program_5 {

    // THE CHAOS ENGINE - Handmade LCG for erratic timing
    private static MyRandom chaosEngine = new MyRandom(System.currentTimeMillis(), false);
    
    // THE AESTHETIC PALETTE (ANSI ESCAPE CODES)
    public static final String RESET      = "\u001B[0m";
    public static final String RED        = "\u001B[31m";
    public static final String CYAN       = "\u001B[36m";
    public static final String PURPLE     = "\u001B[35m";
    public static final String GREY       = "\u001B[90m";
    public static final String WHITE_BOLD = "\u001B[1;37m";

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        String city1 = "", city2 = "", city3 = "";
        String[] finalResults = new String[3];
        
        // THE MASTER STATE MACHINE
        // Implemented via switch statement as requested by GDoc
        int systemState = 0; 
        boolean machineRunning = true;

        while (machineRunning) {
            switch (systemState) {

                    System.out.println(CYAN + "INTERFACING WITH MACINTOSH FRED..." + RESET);

                    systemState = 1;
                    break;

                    System.out.println("\n" + GREY + "--- [SENSORY INPUT INTERFACE] ---" + RESET);
                    city1 = safeReadCity(input, "1. transport(city one):  ");
                    city2 = safeReadCity(input, "2. motorways(city two):  ");
                    city3 = safeReadCity(input, "3. tramlines(city three):  ");

                    if (manualCompare(city1, city2) == 0 || 
                        manualCompare(city2, city3) == 0 || 
                        manualCompare(city1, city3) == 0) {
                        
                        chaosEngine.triggerChaos();
                        panicAndroid();
                    }
                    systemState = 2;
                    break;

                    heartbeat("sorting");

                    heartbeat("output");

                    System.out.println(RED + "This is what you get when you mess with us." + RESET);
                    machineRunning = false;
                    break;
            }
        }
        input.close();
    }

    // =========================================================================
    // THE PATTERN MATRIX (Core Logic)
    // =========================================================================
    
    public static String[] sortThreeCities(String a, String b, String c) {
        String A = toUpper(a);
        String B = toUpper(b);
        String C = toUpper(c);

        // Pattern mapping: combines three comparisons into a single switch key
        // + is greater, - is less, 0 is equal
        String pattern = sign(manualCompare(A, B)) + 
                         sign(manualCompare(A, C)) + 
                         sign(manualCompare(B, C));

        String first, second, third;

        /*
           SWITCH MATRIX:
           Exhaustive mapping of the six possible sorting permutations.
        */
        switch (pattern) {
            case "---": first = a; second = b; third = c; break; // A < B < C
            case "--+": first = a; second = c; third = b; break; // A < C < B
            case "+--": first = b; second = a; third = c; break; // B < A < C
            case "+-+": first = b; second = c; third = a; break; // B < C < A
            case "-+-": first = c; second = a; third = b; break; // C < A < B
            case "-++": first = c; second = b; third = a; break; // C < B < A
            default:
                // Handle equality or edge anomalies
                first = a; second = b; third = c;
                break;
        }

        return new String[] { first, second, third };
    }

    public static void fredSpeak(String message) {
        try {
            // Fitter Happier voice is characterized by Rate -4

                            "$s.Speak('" + message.replace("'", "''") + "');";
            
            new ProcessBuilder("PowerShell", "-Command", script).start();
        } catch (Exception e) {
            // The machine suffers in silence
        }
    }

    public static int manualCompare(String s1, String s2) {
        char[] array1 = s1.toCharArray();
        char[] array2 = s2.toCharArray();
        
        // Manual length counting
        int n1 = 0; for (char c : array1) n1++;
        int n2 = 0; for (char c : array2) n2++;
        
        int minLength = (n1 < n2) ? n1 : n2;

        for (int i = 0; i < minLength; i++) {
            if (array1[i] < array2[i]) return -1;
            if (array1[i] > array2[i]) return 1;
        }

        // Switch to evaluate remaining length
        if (n1 < n2) return -1;
        if (n1 > n2) return 1;
        return 0;
    }

    public static String toUpper(String s) {
        StringBuilder sb = new StringBuilder();
        for (char c : s.toCharArray()) {
            // Manual ASCII conversion logic
            if (c >= 'a' && c <= 'z') {
                sb.append((char)(c - 32));
            } else {
                sb.append(c);
            }
        }
        return sb.toString();
    }

    public static String sign(int x) {
        // Sign mapping for the pattern switch string
        switch (x) {
            case -1: return "-";
            case 1:  return "+";
            default: return "0";
        }
    }

    public static void heartbeat(String phase) {
        String tag;
        // Selection of cinematic phase labels
        switch (phase) {
            case "boot":    tag = "AIRBAG"; break;
            case "sorting": tag = "LET_DOWN"; break;
            case "output":  tag = "LUCKY";  break;
            default:        tag = "STATIC"; break;

        System.out.print(GREY + "[" + tag + "] " + RESET);

    }

    public static void panicAndroid() {

    public static String safeReadCity(Scanner input, String prompt) {
        while (true) {
            System.out.print(prompt);
            if (!input.hasNextLine()) {
                fredSpeak("disappearing completely.");
                System.exit(0);
            }
            
            String raw = input.nextLine();
            MiniTokenizer st = new MiniTokenizer(raw);
            StringBuilder sb = new StringBuilder();
            
            String token = st.nextToken();
            while (token != null) {
                if (isAlphanumeric(token)) {
                    if (sb.length() > 0) sb.append(" ");
                    sb.append(token);
                }
                token = st.nextToken();
            }
            
            String result = sb.toString().trim();
            if (result.length() > 0) return result;
            
            System.out.println(RED + "Input rejected. Try again." + RESET);
        }
    }

    public static boolean isAlphanumeric(String s) {
        if (s == null || s.length() == 0) return false;
        char c = s.charAt(0);
        // Manual range check for alphabet and numerals
        boolean isLower = (c >= 'a' && c <= 'z');
        boolean isUpper = (c >= 'A' && c <= 'Z');
        boolean isNum   = (c >= '0' && c <= '9');
        return isLower || isUpper || isNum;
    }

        System.out.println(GREY + "   '''''  " + RESET);
        System.out.println(CYAN + "[END OF LINE]" + RESET);
    }

    static class MyRandom {
        private long seed;
        private boolean chaos;

        MyRandom(long seed, boolean chaos) {
            this.seed = seed ^ 0x5DEECE66DL;
            this.chaos = chaos;
        }

        void triggerChaos() { this.chaos = true; }
        boolean isChaos() { return this.chaos; }

        int nextInt(int bound) {
            if (bound <= 0) return 0;
            seed = (seed * 1664525 + 1013904223) % (1L << 32);
            int value = (int)(seed % bound);
            if (value < 0) value = -value;
            return chaos ? (value * 13 + 7) % bound : value;
        }
    }

    static class MiniTokenizer {
        private final String input;
        private int index = 0;

        MiniTokenizer(String input) {
            this.input = input;
        }

        String nextToken() {
            // Manual space skipping logic
            while (index < input.length() && input.charAt(index) <= ' ') {
                index++;
            }

            if (index >= input.length()) return null;

            char c = input.charAt(index);
            boolean alnum = (c >= 'a' && c <= 'z') || 
                            (c >= 'A' && c <= 'Z') || 
                            (c >= '0' && c <= '9');

            if (!alnum) {
                index++;
                return "" + c;
            }

            StringBuilder word = new StringBuilder();
            while (index < input.length()) {
                char current = input.charAt(index);
                boolean currentAlnum = (current >= 'a' && current <= 'z') || 
                                       (current >= 'A' && current <= 'Z') || 
                                       (current >= '0' && current <= '9');
                if (!currentAlnum) break;
                word.append(current);
                index++;
            }
            return word.toString();
        }
    }
